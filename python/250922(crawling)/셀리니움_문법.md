# 3. 체크박스 <input type='checkbox' id='agree_all_terms>
```
agree_checkbox = driver.find_element(By.ID,'agree_all_terms')
# 현재 체크가 안 되어 있을 경우만 클릭
if not agree_checkbox.is_selected():
    agree_checkbox.click()
```
